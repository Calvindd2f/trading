# This code demonstrates how to save and load a DataFrame using joblib.

# Example usage:
# First, create a random DataFrame with 10 rows and 5 columns
np.random.seed(0)  # for reproducibility
df = pd.DataFrame(np.random.rand(10, 5))

# Save the DataFrame to disk using joblib's dump function
dump(df, 'df.joblib')

# Load the DataFrame from disk using joblib's load function
loaded_df = load('df.joblib')

# Print the original and loaded DataFrames to verify that they are identical
print("Original DataFrame:")
print(df)
print("\nLoaded DataFrame:")
print(loaded_df)
